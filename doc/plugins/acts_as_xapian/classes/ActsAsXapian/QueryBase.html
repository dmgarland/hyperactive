319:             end
320: 
321:             # Look up without too many SQL queries
322:             lhash = {}
323:             lhash.default = []
324:             for doc in docs

326:                 lhash[k[0]] = lhash[k[0]] + [k[1]]
327:             end
328:             # for each class, look up all ids
329:             chash = {}
330:             for cls, ids in lhash
331:                 conditions = [ "#{cls.constantize.table_name}.#{cls.constantize.primary_key} in (?)", ids ]
332:                 found = cls.constantize.find(:all, :conditions => conditions, :include => cls.constantize.xapian_options[:eager_load])

334:                     chash[[cls, f.id]] = f
335:                 end
336:             end
